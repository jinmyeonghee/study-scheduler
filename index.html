function downloadCSV() { const rows = [['unit_id', 'type', 'day1', 'day1_time', 'day2', 'day2_time', 'day3', 'day3_time']]; let totalDays = 0; summaryData.forEach(row => { const unitId = row.unit; const type = getUnitType(unitId); if (type === 'day1') totalDays += 1; else if (type === 'day2') totalDays += 2; else if (type === 'day3') totalDays += 3; rows.push([ unitId, type, getDayText(row, 'd1'), getDayTime(row, 'd1').toFixed(1), getDayText(row, 'd2'), getDayTime(row, 'd2').toFixed(1), getDayText(row, 'd3'), getDayTime(row, 'd3').toFixed(1) ]); }); // 👇 요약 (딱 2개만) rows.push([]); rows.push(['요약']); rows.push([ `총 예상 소요일: ${totalDays}일`, `하루 기준 시간: ${getDailyLimit()}분` ]); const csv = '\ufeff' + rows.map(row => row.map(cell => `"${String(cell).replace(/"/g, '""')}"`).join(',') ).join('\n'); const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' }); const url = URL.createObjectURL(blob); const link = document.createElement('a'); link.href = url; link.download = 'study_schedule.csv'; link.click(); URL.revokeObjectURL(url); }
