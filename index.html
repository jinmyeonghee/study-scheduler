function downloadCSV() { const rows = [['unit_id', 'type', 'day1', 'day1_time', 'day2', 'day2_time', 'day3', 'day3_time', 'missing']]; let day1Count = 0; let day2Count = 0; let day3Count = 0; let totalDays = 0; let overCount = 0; let missingCount = 0; summaryData.forEach(row => { const unitId = row.unit; const type = getUnitType(unitId); const activeKeys = getActiveKeysForUnit(row); const assigned = Object.keys(state[unitId] || {}); const missing = activeKeys.filter(k => !assigned.includes(k)).map(labelName); if (type === 'day1') { day1Count++; totalDays += 1; } else if (type === 'day2') { day2Count++; totalDays += 2; } else if (type === 'day3') { day3Count++; totalDays += 3; } if (missing.length > 0) missingCount++; ['d1','d2','d3'].forEach(day => { const time = getDayTime(row, day); if (time > getDailyLimit()) overCount++; }); rows.push([ unitId, type, getDayText(row, 'd1'), getDayTime(row, 'd1').toFixed(1), getDayText(row, 'd2'), getDayTime(row, 'd2').toFixed(1), getDayText(row, 'd3'), getDayTime(row, 'd3').toFixed(1), missing.join(' / ') ]); }); rows.push([]); rows.push([ '커리큘럼 요약', '', `총 예상 소요일: ${totalDays}일`, '', `day1 소단원: ${day1Count}개`, '', `day2 소단원: ${day2Count}개`, '', `day3 소단원: ${day3Count}개` ]); rows.push([ '', '', `기준시간 초과: ${overCount}건`, '', `미완성 소단원: ${missingCount}개`, '', `하루 기준 시간: ${getDailyLimit()}분`, '', '' ]); const csv = '\ufeff' + rows.map(row => row.map(cell => `"${String(cell).replace(/"/g, '""')}"`).join(',') ).join('\n'); const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' }); const url = URL.createObjectURL(blob); const link = document.createElement('a'); link.href = url; link.download = 'study_schedule.csv'; link.click(); URL.revokeObjectURL(url); }
